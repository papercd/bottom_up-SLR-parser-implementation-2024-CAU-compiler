CODE -> CODE_BLOCK
CODE -> ''

CODE_BLOCK -> VDECL CODE_BLOCK
CODE_BLOCK -> FDECL CODE_BLOCK
CODE_BLOCK -> STMT CODE_BLOCK

VDECL -> vtype id vdecl_tail semi

vdecl_tail -> ''
vdecl_tail -> ASSIGN

ASSIGN -> id assign RHS

RHS -> EXPR
RHS -> literal
RHS -> character
RHS -> boolstr

EXPR -> EXPR addsub TERM
EXPR -> TERM

TERM -> TERM multdiv FACTOR
TERM -> FACTOR

FACTOR -> lparen EXPR rparen
FACTOR -> id
FACTOR -> num

FDECL -> vtype id lparen ARG rparen lbrace BLOCK RETURN rbrace

ARG -> ARG_LIST
ARG -> ''

ARG_LIST -> vtype id
ARG_LIST -> vtype id comma ARG_LIST

BLOCK -> STMT BLOCK
BLOCK -> ''

STMT -> VDECL semi
STMT -> ASSIGN semi
STMT -> IF_STMT
STMT -> WHILE_STMT

IF_STMT -> if lparen COND rparen lbrace BLOCK rbrace ELSE_STMT

ELSE_STMT -> else if lparen COND rparen lbrace BLOCK rbrace ELSE_STMT
ELSE_STMT -> else lbrace BLOCK rbrace

WHILE_STMT -> while lparen COND rparen lbrace BLOCK rbrace

COND -> EXPR comp EXPR
COND -> boolstr

RETURN -> return RHS semi
